有7只猴子平分一堆香蕉，第一只猴子平分为7份之后发现还多1根，于是他拿走了多余的一根以及7份中的一份；   
第二只猴子像第一只猴子一样，从剩下的香蕉中拿走多余一根，剩下的恰好能分成7份。  
后面几只猴子都是这样，请问这里面一共有多少根香蕉？

```python
for banana in range(1, 9999999):
    if banana % 7 == 1:  # 第一只猴子平分为7份之后发现还多1根
        left1 = banana - (banana - 1) / 7 - 1  # 剩下的香蕉 6
        if left1 % 7 == 1:  # 第二只猴子像第一只猴子一样，从剩下的香蕉中拿走多余一根
            left2 = left1 - (left1 - 1) / 7 - 1  # 5
            if left2 % 7 == 1:  # 3
                left3 = left2 - (left2 - 1) / 7 - 1  # 4
                if left3 % 7 == 1:  # 4
                    left4 = left3 - (left3 - 1) / 7 - 1  # 3
                    if left4 % 7 == 1:  # 5
                        left5 = left4 - (left4 - 1) / 7 - 1  # 2
                        if left5 % 7 == 1:  # 6
                            left6 = left5 - (left5 - 1) / 7 - 1  # 1
                            if left6 % 7 == 1:
                                xj = banana
                                print(f"香蕉一共有{xj}根")
                                # 1 M
                                m1 = (xj - 1)/7
                                print(f"第一只猴子拿走了{int(m1) + 1}根;",end='')
                                m1s = xj - m1 - 1
                                # 2 M
                                m2 = (m1s - 1) / 7
                                print(f"第二只猴子拿走了{int(m2) + 1}根;",end='')
                                m2s = m1s - m2 - 1

                                m3 = (m2s - 1) / 7
                                print(f"第三只猴子拿走了{int(m3) + 1}根;",end='')
                                m3s = m2s - m3 - 1
                                # 4 M
                                m4 = (m3s - 1) / 7
                                print(f"第四只猴子拿走了{int(m4) + 1}根;",end='')
                                m4s = m3s - m4 - 1
                                # 5 M
                                m5 = (m4s - 1) / 7
                                print(f"第五只猴子拿走了{int(m5) + 1}根;",end='')
                                m5s = m4s - m5 - 1
                                # 6 M
                                m6 = (m5s - 1) / 7
                                print(f"第六只猴子拿走了{int(m6) + 1}根;",end='')
                                m6s = m5s - m6 - 1
                                # 6 M
                                m7 = (m6s - 1) / 7
                                print(f"第七只猴子拿走了{int(m7) + 1}根")
                                print()
```

```python
monkey = int(input("请输入猴子的数量："))

#定义香蕉总数函数
def shown(n):
    #循环次数
    for i in range(1, monkey + 1):
        #下一只猴子应该带走的香蕉数
        t = (n - 1) / monkey
        #格式化输出
        print(f'{i}. 第{i}只猴吃1个, 拿走{t}个。')
        #前一只猴子带走一份香蕉后剩余的香蕉总数
        n = (monkey-1) * t

def main():
    #从1开始
    k = 1
    while True:
        t = k
        #循环次数
        for _ in range(monkey-1):
            #当前猴子应拿走香蕉数为tc，吃拿之前总量应为 monkey * tc + 1,前一个猴子拿走香蕉数为tp，则有 (monkey-1) * tp = monkey * tc + 1
            t = monkey * t + 1
            #在for循环中含有break时则直接终止循环，不执行else
            if t % (monkey-1): break
            t /= (monkey-1)
        #当迭代的对象迭代完并为空时，位于else的子句将执行,即找到符合条件最小整数
        else:
             print(f'如果猴子{monkey}只:')
             print(f'香蕉总数要{(monkey * t + 1)}个')
             shown(monkey * t + 1)
             break
        k += 1
main()
```
